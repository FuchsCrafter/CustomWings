## Maven Dependency:



	<repositories>
		<repository>
		    <id>jitpack.io</id>
		    <url>https://jitpack.io</url>
		</repository>
	</repositories>

---

    <dependency>
	    <groupId>com.github.YouHaveTrouble</groupId>
	    <artifactId>CustomWings</artifactId>
	    <version>Tag</version>
	</dependency>
	
## Gradle:

	allprojects {
		repositories {
			...
			maven { url 'https://jitpack.io' }
		}
	}
	
---
	
	dependencies {
    	        implementation 'com.github.YouHaveTrouble:CustomWings:Tag'
    	}
